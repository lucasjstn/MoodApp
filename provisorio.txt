  Text,

const App = () => {

  const [text, setText] = useState("Qualquer Texto")

  const aoClicarMudarTexto = () => {
    setText("O texto foi alterado pois o bot√£o foi clicado.");

  }
 return (
   <View style={styles.body}>

   </View>
 );
}
